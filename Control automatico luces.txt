Serial.begin(9600);
IrReceiver.begin(IR, DISABLE_LED_FEEDBACK);

pinMode(LED3, OUTPUT);
pinMode(LED4, OUTPUT);
}

del LED2

(IrReceiver.decodedIRData.decodedRawData ==
Tecla_4) {

del LED3

} else if
(IrReceiver.decodedIRData.decodedRawData ==
Tecla_5){
estadoLED4 = !estadoLED4; // Invierte el estado
del LED3

}

}
